\section{Mandatory Preliminary Checks}

Before any money transfer, you must absolutely ensure the legitimacy and reliability of the recipient association.

\subsection{Clearly Identify the Association}
\begin{checklist}
    \item Exact name of the association (as officially registered)
    \item Country of operation (may differ from the country of registration)
    \item Main purpose: health, education, environment, human rights, etc.
    \item Year of establishment and tenure of leaders
\end{checklist}

\subsection{Verify Legality in its Home Country}
\begin{checklist}
    \item Is the association registered with the relevant authorities?
    \begin{itemize}
        \item In Francophone Africa: ministerial receipt (Interior or Territorial Administration)
        \item In Anglophone Africa: certificate of incorporation
        \item In Lusophone Africa: registro da associação
    \end{itemize}
    \item Does it have a Tax Identification Number (TIN/NIF)?
    \item Is it up-to-date with its annual declarations?
\end{checklist}

\subsection{Look for Reviews and Transparency Reports}
\begin{checklist}
    \item Consult the association's annual reports (at least the last two fiscal years)
    \item Verify its presence on recognized platforms:
    \begin{itemize}
        \item GiveWell (recommendation based on demonstrated effectiveness)
        \item GuideStar (financial transparency)
        \item GlobalGiving (verified projects)
    \end{itemize}
    \item Search for press articles or case studies involving the association
\end{checklist}

\subsection{Contact the Association Directly}
\begin{arrowlist}
    \item Send an email or call before giving
    \item Ask specific questions
    \item Assess the quality and speed of their response
\end{arrowlist}

\begin{helpbox}
    \centering
    \info\ \textbf{Good to know} \\
    A reliable and well-structured association will respond within a reasonable time (48 to 72 hours) and will provide the requested information without difficulty.
\end{helpbox}

\section{Choosing the Transfer Method}

\subsection{Comparison of Transfer Methods}

\begin{tcolorbox}[colback=lightgray, colframe=darkgray, title=Comparison Table]
    \small
    \renewcommand{\arraystretch}{1.5}
    \begin{tabular}{|p{3cm}|p{2cm}|p{2cm}|p{2.5cm}|p{2cm}|}
        \hline
        \textbf{Method} & \textbf{Fees} & \textbf{Time} & \textbf{Security} & \textbf{Recommendation} \\
        \hline
        SWIFT Transfer & 20-50 € + exchange & 2-5 days & High & Excellent \\
        \hline
        Wise (TransferWise) & 0.5-1\% & 1-2 days & High & Excellent \\
        \hline
        Revolut & 0-1\% & Instant & High & Very Good \\
        \hline
        Platform (GlobalGiving) & 5-15\% & Variable & High & Good \\
        \hline
        Mobile Money & 1-5\% + exchange & Instant & Medium & Medium \\
        \hline
        Western Union & 5-15\% & Instant & Low & Avoid \\
        \hline
        Cash & 0\% & Instant & Very Low & Avoid \\
        \hline
    \end{tabular}
\end{tcolorbox}

\subsection{Details by Transfer Method}

\subsubsection{International Bank Transfer (SWIFT)}
\begin{arrowlist}
    \item \textbf{To request from the association:} IBAN, BIC/SWIFT code
    \item \textbf{Verification:} The account holder's name must be the association's name
    \item \textbf{Note:} Intermediary fees may apply
\end{arrowlist}

\subsubsection{SEPA Transfer}
\begin{arrowlist}
    \item \textbf{Condition:} Account in a SEPA zone bank
    \item \textbf{Advantage:} Generally zero fees
\end{arrowlist}

\subsubsection{Secure Donation Platforms}
\begin{arrowlist}
    \item GlobalGiving: verified projects
    \item HelloAsso: for projects carried by French associations
\end{arrowlist}

\begin{alertbox}
    \centering
    \textbf{Methods to absolutely avoid} \\
    \vspace{0.2cm}
    Western Union, sending cash, personal bank account
\end{alertbox}

% --- Section 3 ---
\section{Optimizing Fees and Exchange Rates}

\subsection{Compare Bank Fees}
\begin{starredlist}
    \item Check your bank's fee schedule
    \item Watch out for hidden fees (fixed commission + exchange fees)
\end{starredlist}

\subsection{Use a Low-Cost Transfer Service}
\begin{starredlist}
    \item \textbf{Wise:} Recommended, transparent fees (0.5-1\%)
    \item \textbf{Revolut:} Very competitive
\end{starredlist}

\subsection{Choose the Sending Currency}
\begin{starredlist}
    \item Ask the association which currency is most convenient
    \item Sending in Euros or Dollars is often advantageous
\end{starredlist}

\section{Securing Your Donation}

\begin{checklist}
    \item \textbf{Never donate without a receipt}
    \item \textbf{Demand an official document}
    \item \textbf{Protect your personal data}
    \item \textbf{Report any suspicious behavior}
\end{checklist}

\subsection{Tax Receipt and Certificate}
\begin{arrowlist}
    \item The association must provide a certificate including:
    \begin{itemize}
        \item Its name, address, and registration number
        \item Your name and address
        \item The date and amount
    \end{itemize}
\end{arrowlist}

% --- Section 5 ---
\section{Tax Benefits by Country}

\subsection{France}
\begin{heartlist}
    \item Rate: 66\% of the donation (limit of 20\% of income)
    \item The association must have a fiscal relay in France
\end{heartlist}

\subsection{Belgium}
\begin{heartlist}
    \item Rate: 45\% of the donation (minimum 40 €)
    \item The association must be recognized
\end{heartlist}

\subsection{Switzerland}
\begin{heartlist}
    \item Rate varies by canton (20-30\%)
\end{heartlist}
